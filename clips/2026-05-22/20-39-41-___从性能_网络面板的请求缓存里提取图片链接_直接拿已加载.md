// 从性能/网络面板的请求缓存里提取图片链接（直接拿已加载的）
const urls = [];
// 遍历所有请求缓存
performance.getEntriesByType('resource').forEach(req => {
  const url = req.name;
  // 只保留原图，过滤掉缩略图和非图片请求
  if (/\.png|\.jpg|\.jpeg$/.test(url) && !url.includes('thumb')) {
    urls.push(url);
  }
});

console.log(`✅ 从已加载请求中提取到 ${urls.length} 张原图链接：`);
console.log(urls.join('\n'));
console.log('复制上面所有链接，用wget或下载工具批量下载即可');
