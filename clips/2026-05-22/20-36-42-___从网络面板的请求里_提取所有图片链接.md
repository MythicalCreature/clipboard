// 从网络面板的请求里，提取所有图片链接
const urls = Array.from(document.querySelectorAll('img'))
  .map(img => img.src)
  .filter(src => /\.(png|jpg|jpeg)$/.test(src) && !src.includes('thumb')); // 过滤掉缩略图，保留原图

console.log('所有原图链接（共' + urls.length + '张）：');
console.log(urls.join('\n'));
console.log('复制上面所有链接，用批量下载工具一键下载即可');
