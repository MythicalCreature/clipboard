// 批量导出摹客页面所有图片为PNG
(async function() {
  // 1. 获取页面所有img元素
  const imgs = Array.from(document.querySelectorAll('img'));
  console.log(`找到 ${imgs.length} 张图片`);

  // 2. 过滤出有效的图片（排除空白、占位图）
  const validImgs = imgs.filter(img => {
    const src = img.src || img.currentSrc;
    return src && src.startsWith('http') && !src.includes('data:image/gif;base64,R0lGODlhAQABAAAAACH5BAEKAAEALAAAAAABAAEAAAICTAEAOw==');
  });

  if (validImgs.length === 0) {
    alert('没有找到有效图片，请刷新页面后再试');
    return;
  }

  // 3. 批量下载并转换为PNG
  for (let i = 0; i < validImgs.length; i++) {
    const img = validImgs[i];
    const url = img.src || img.currentSrc;
    const filename = `mockplus_img_${String(i+1).padStart(3, '0')}.png`;

    try {
      // 创建canvas将图片转为PNG
      const canvas = document.createElement('canvas');
      const ctx = canvas.getContext('2d');
      const image = new Image();
      image.crossOrigin = 'anonymous'; // 解决跨域问题
      await new Promise((resolve, reject) => {
        image.onload = () => {
          canvas.width = image.naturalWidth;
          canvas.height = image.naturalHeight;
          ctx.drawImage(image, 0, 0);
          resolve();
        };
        image.onerror = reject;
        image.src = url;
      });

      // 下载PNG
      const link = document.createElement('a');
      link.download = filename;
      link.href = canvas.toDataURL('image/png');
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);

      console.log(`✅ 已下载: ${filename}`);
      await new Promise(resolve => setTimeout(resolve, 300)); // 防止浏览器拦截批量下载
    } catch (err) {
      console.error(`❌ 下载失败: ${url}`, err);
    }
  }

  alert(`全部处理完成！成功下载 ${validImgs.length} 张图片`);
})();
