// 从当前已加载的网络请求里，提取不带缩略图参数的原图链接
performance.getEntriesByType('resource')
  .map(req => req.name)
  .filter(url => {
    // 匹配摹客图片链接格式，同时过滤掉带 thumb 参数的请求
    return /\.png/.test(url) && !url.includes('thumb=') && !url.includes('thumb-p-');
  })
  .forEach(url => console.log(url));
