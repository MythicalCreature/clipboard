// 获取当前项目ID和画板ID
const projectId = window.location.pathname.split('/')[3];
const artboardIds = Array.from(document.querySelectorAll('.artboard-item'))
  .map(el => el.dataset.id)
  .join(',');
console.log(`项目ID: ${projectId}`);
console.log(`所有画板ID: ${artboardIds}`);
