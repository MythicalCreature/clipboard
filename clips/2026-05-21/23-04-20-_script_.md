  <script>

        let suites = [];
        let allQuestions = [];
        const STORE_KEY = "xsyy_verbal_suite_answers_v1";
        let progress = JSON.parse(localStorage.getItem(STORE_KEY) || "{}");
        let currentSuite = "all";
        let reveal = false;

        console.log("=== 开始加载 ===");

        (async function autoLoad() {
            try {
                // 1. 读取 files.json
                const res = await fetch("files.json");
                const fileList = await res.json();
                console.log("✅ files.json 读取成功，文件数：", fileList.length);

                const allLoadedData = [];
                // 2. 逐个加载 JSON
                for (const file of fileList) {
                    try {
                        const r = await fetch(file);
                        if (!r.ok) throw new Error("状态码" + r.status);
                        const data = await r.json();
                        allLoadedData.push({ file, data });
                        console.log("✅ 加载成功：", file, "数据类型：", Array.isArray(data) ? "数组" : "对象");
                    } catch (e) {
                        console.error("❌ 加载失败：", file, e);
                    }
                }

                console.log("=== 解析数据 ===");
                const suiteList = [];
                for (const { file, data } of allLoadedData) {
                    if (Array.isArray(data)) {
                        // 格式A：数组里包含多个套题对象
                        for (const item of data) {
                            if (item && item.name && (item.question || item.questions)) {
                                suiteList.push({
                                    name: item.name,
                                    questions: item.question || item.questions || []
                                });
                            }
                        }
                    } else if (data && data.name && (data.question || data.questions)) {
                        // 格式B：单个套题对象
                        suiteList.push({
                            name: data.name,
                            questions: data.question || data.questions || []
                        });
                    }
                }

                console.log("✅ 解析完成，套题总数：", suiteList.length);
                allQuestions = suiteList.flatMap(s => s.questions);
                console.log("✅ 题目总数：", allQuestions.length);

                // 生成 suites 数组
                suites = suiteList.map((item, idx) => ({
                    suiteId: idx + 1,
                    suiteName: item.name,
                    count: item.questions.length
                }));
                console.log("✅ suites 生成完成：", suites);

                // 调用 render()
                console.log("=== 调用 render() ===");
                render();
                console.log("=== 全部加载完成 ===");

                // 👇 关键修复：手动更新页面统计和列表
                const totalInfoEl = document.querySelector(".total-info");
                if (totalInfoEl) {
                    totalInfoEl.textContent = `共 ${suites.length} 套，${allQuestions.length} 道选择题。答案会自动保存在本机浏览器里；可导出/导入进度，方便换设备继续。`;
                }

                const suiteListEl = document.querySelector(".suite-list");
                if (suiteListEl) {
                    suiteListEl.innerHTML = suites.map(suite => `
        <div class="suite-item" data-id="${suite.suiteId}">
          ${suite.suiteName} (${suite.count}题)
        </div>
      `).join("");
                }

            } catch (err) {
                console.error("❌ 加载过程出错：", err);
            }
        })();

        // 自动统计总套数、总题数（零报错版）
        const totalSuites = suites.length;
        const totalQuestions = suites.reduce((sum, s) => sum + s.questions.length, 0);

        // 安全更新页面数字（不会报错）
        try {
            document.getElementById("totalInfo").textContent = `共 ${totalSuites} 套，${totalQuestions} 道选择题。答案会自动保存在本机浏览器里；可导出/导入进度，方便换设备继续。`;
            document.getElementById("totalCount").textContent = totalQuestions;
        } catch (e) { }

        const suiteList = document.getElementById("suiteList");
        const suiteSelect = document.getElementById("suiteSelect");
        const questionsEl = document.getElementById("questions");

        function save() {
            localStorage.setItem(STORE_KEY, JSON.stringify(progress));
        }

        function normAnswer(v) {
            return String(v || "").toUpperCase().replace(/[^A-Z]/g, "").split("").sort().join("");
        }

        function isMulti(q) {
            return normAnswer(q.answer).length > 1;
        }

        function getUserAnswer(id) {
            return normAnswer(progress[id]?.answer || "");
        }

        function isCorrect(q) {
            const ua = getUserAnswer(q.id);
            if (!ua)
                return null;
            const ca = normAnswer(q.answer);
            if (!ca)
                return null;
            return ua === ca;
        }

        function initSuites() {
            // 下拉框：用 题目里的 suiteIndex 作为 value
            suiteSelect.innerHTML = '<option value="all">全部套题</option>' + suites.map((s) => {
                // 取这套题第一题的 suiteIndex
                const suiteIndex = s.questions[0]?.suiteIndex || 0;
                return `<option value="${suiteIndex}">${escapeHtml(s.name)}（${s.questions.length}）</option>`;
            }).join("");

            // 侧边按钮：用 题目里的 suiteIndex 作为 data-suite
            suiteList.innerHTML = suites.map((s) => {
                // 取这套题第一题的 suiteIndex
                const suiteIndex = s.questions[0]?.suiteIndex || 0;
                return `<button class="suite-btn" data-suite="${suiteIndex}">
            ${escapeHtml(s.name)} 
            <span class="suite-count">${s.questions.length}</span>
        </button>`;
            }).join("");

            // 点击事件
            document.querySelectorAll(".suite-btn").forEach(btn => {
                btn.addEventListener("click", () => {
                    currentSuite = btn.dataset.suite;
                    suiteSelect.value = currentSuite;
                    render();
                });
            });

            suiteSelect.addEventListener("change", () => {
                currentSuite = suiteSelect.value;
                render();
            });
        }

        function escapeHtml(s) {
            return String(s).replace(/[&<>"']/g, m => ({
                '&': '&amp;',
                '<': '&lt;',
                '>': '&gt;',
                '"': '&quot;',
                "'": '&#39;'
            }[m]));
        }

        function optionLabel(idx) {
            return String.fromCharCode(65 + idx);
        }

        function optionText(raw, idx) {
            let t = String(raw || "").trim();
            if (/^[A-Z][.．、\s]/.test(t))
                return t;
            return optionLabel(idx) + ". " + t;
        }

        function filteredQuestions() {
            const keyword = document.getElementById("search").value.trim().toLowerCase();
            const filter = document.getElementById("filter").value;
            let list = allQuestions.filter(q => currentSuite === "all" || String(q.suiteIndex) === String(currentSuite));
            if (keyword) {
                list = list.filter(q => (q.suiteName + " " + q.stem + " " + q.options.join(" ")).toLowerCase().includes(keyword));
            }
            list = list.filter(q => {
                const ua = getUserAnswer(q.id);
                const ok = isCorrect(q);
                if (filter === "unanswered")
                    return !ua;
                if (filter === "answered")
                    return !!ua;
                if (filter === "wrong")
                    return ok === false;
                if (filter === "correct")
                    return ok === true;
                if (filter === "doubt")
                    return !!progress[q.id]?.doubt;
                return true;
            }
            );
            return list;
        }

        function render() {
            const filter = document.getElementById("filter").value;

            // 👇 👇 👇 最终完美逻辑
            if (filter === "wrong") {
                // 只有【刚刚切换筛选】时才自动展开
                // 手动点过隐藏的，不强制覆盖
                if (window.lastFilter !== filter) {
                    reveal = true;
                }
            }
            // 记录上一次的筛选状态
            window.lastFilter = filter;

            document.querySelectorAll(".suite-btn").forEach(btn => btn.classList.toggle("active", btn.dataset.suite === String(currentSuite)));
            suiteSelect.value = currentSuite;
            const list = filteredQuestions();
            let done = 0
                , correct = 0
                , wrong = 0
                , doubt = 0;
            list.forEach(q => {
                const ua = getUserAnswer(q.id);
                const ok = isCorrect(q);
                if (ua)
                    done++;
                if (ok === true)
                    correct++;
                if (ok === false)
                    wrong++;
                if (progress[q.id]?.doubt)
                    doubt++;
            }
            );
            document.getElementById("statTotal").textContent = list.length;
            document.getElementById("statDone").textContent = done;
            document.getElementById("statCorrect").textContent = correct;
            document.getElementById("statWrong").textContent = wrong;
            document.getElementById("statDoubt").textContent = doubt;
            questionsEl.innerHTML = list.map(renderQuestion).join("") || '<div class="card">没有符合条件的题目。</div>';
            bindQuestionEvents();
        }

        function renderQuestion(q) {
            const user = getUserAnswer(q.id);
            const ok = isCorrect(q);
            const result = ok === null ? "" : `<span class="result ${ok ? "ok" : "bad"}">${ok ? "✓ 正确" : "✗ 错误"}</span>`;
            const isDoubt = !!progress[q.id]?.doubt;
            const type = isMulti(q) ? "checkbox" : "radio";
            const selected = new Set(user.split(""));
            const opts = q.options.map((o, idx) => {
                const label = optionLabel(idx);
                return `<label class="opt">
      <input type="${type}" name="${q.id}" value="${label}" ${selected.has(label) ? "checked" : ""}>
      <span>${escapeHtml(optionText(o, idx))}</span>
    </label>`;
            }
            ).join("");
            const analysis = q.analysis ? `<div class="analysis" style="${reveal ? "display:block" : ""}">解析：${q.analysis}</div>` : "";
            return `<section class="card" id="${q.id}">
    <div class="q-head">
      <div>
        <span class="tag">${escapeHtml(q.suiteName)}</span> ${isDoubt ? `<span class="tag doubt">疑问题</span>` : ""}
        <div class="q-title">${q.num}. ${escapeHtml(q.stem)}</div>
      </div>
      <div class="small">${isMulti(q) ? "多选记录" : "单选"}</div>
    </div>
    <div class="options">${opts}</div>
    <div class="answer-row">
      ${result}
      <button class="doubt-btn ${isDoubt ? "marked" : ""}" data-doubt="${q.id}">${isDoubt ? "取消疑问" : "标记疑问"}</button>
      <button class="secondary show-one" data-id="${q.id}">查看答案</button>
      <span class="answer-pill" style="${reveal ? "display:inline-flex" : ""}">答案：${escapeHtml(q.answer || "未识别")}</span>
      ${analysis}
      <textarea class="note" data-id="${q.id}" placeholder="错因/笔记，可不填">${escapeHtml(progress[q.id]?.note || "")}</textarea>
    </div>
  </section>`;
        }

        function bindQuestionEvents() {
            questionsEl.querySelectorAll('input[type="radio"], input[type="checkbox"]').forEach(inp => {
                inp.addEventListener("change", e => {
                    const id = inp.name;
                    const checked = [...questionsEl.querySelectorAll(`input[name="${id}"]:checked`)].map(x => x.value).sort().join("");

                    // 记录之前是否已有答案
                    const hadAnswer = !!progress[id]?.answer;

                    // 保存答案
                    progress[id] = progress[id] || {};
                    progress[id].answer = checked;
                    save();

                    // 核心逻辑：
                    // 如果是【第一次做题】→ 自动显示答案和解析
                    // 如果是【修改已做题】→ 不自动显示
                    if (!hadAnswer) {
                        setTimeout(() => {
                            const card = document.getElementById(id);
                            if (card) {
                                card.querySelectorAll(".answer-pill,.analysis").forEach(x => x.style.display = "block");
                            }
                        }, 10);
                    }

                    render();
                });
            }
            );
            questionsEl.querySelectorAll(".note").forEach(t => {
                t.addEventListener("input", () => {
                    const id = t.dataset.id;
                    progress[id] = progress[id] || {};
                    progress[id].note = t.value;
                    save();
                }
                );
            }
            );
            questionsEl.querySelectorAll("[data-doubt]").forEach(btn => {
                btn.addEventListener("click", () => {
                    const id = btn.dataset.doubt;
                    progress[id] = progress[id] || {};
                    progress[id].doubt = !progress[id].doubt;
                    save();
                    render();
                }
                );
            }
            );
            questionsEl.querySelectorAll(".show-one").forEach(btn => {
                btn.addEventListener("click", () => {
                    const card = document.getElementById(btn.dataset.id);
                    card.querySelectorAll(".answer-pill,.analysis").forEach(x => x.style.display = "block");
                }
                );
            }
            );
        }

        document.getElementById("search").addEventListener("input", render);
        document.getElementById("filter").addEventListener("change", render);
        document.getElementById("checkAll").addEventListener("click", () => {
            reveal = true;
            render();
        }
        );
        document.getElementById("hideAnswers").addEventListener("click", () => {
            reveal = false;
            render();
        }
        );
        document.getElementById("resetProgress").addEventListener("click", () => {
            if (confirm("确定清空所有作答和笔记吗？")) {
                progress = {};
                save();
                render();
            }
        }
        );
        document.getElementById("exportProgress").addEventListener("click", () => {
            const blob = new Blob([JSON.stringify({
                exportedAt: new Date().toISOString(),
                progress
            }, null, 2)], {
                type: "application/json"
            });
            const a = document.createElement("a");
            a.href = URL.createObjectURL(blob);
            a.download = "言语套题刷题进度.json";
            a.click();
            URL.revokeObjectURL(a.href);
        }
        );
        document.getElementById("importBtn").addEventListener("click", () => document.getElementById("importFile").click());
        document.getElementById("importFile").addEventListener("change", async e => {
            const file = e.target.files[0];
            if (!file)
                return;
            const obj = JSON.parse(await file.text());
            progress = obj.progress || obj || {};
            save();
            render();
        }
        );

        const toggleTopToolsBtn = document.getElementById("toggleTopTools");
        if (toggleTopToolsBtn) {
            toggleTopToolsBtn.addEventListener("click", () => {
                document.body.classList.toggle("top-tools-collapsed");
                toggleTopToolsBtn.textContent = document.body.classList.contains("top-tools-collapsed") ? "展开顶部工具栏" : "收起顶部工具栏，专心刷题";
            }
            );
        }

        initSuites();
        render();
    </script>
