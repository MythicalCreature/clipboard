// 【新增】单独请求单题考点的函数（适配你真实的接口返回格式）
async function getQuestionKeyPoints(questionId) {
  try {
    const url = `https://tiku.fenbi.com/api/xingce/solution/keypoints?ids=${questionId}&app=web&kav=125&av=127&hav=125&version=3.0.0.0&deviceId=&gav=2&apcId=0`;
    
    const res = await fetch(url, {
      "headers": {
        "accept": "application/json, text/plain, */*",
        "accept-language": "zh-CN,zh;q=0.9,en;q=0.8,en-GB;q=0.7,en-US;q=0.6",
        "priority": "u=1, i",
        "sec-ch-ua": "\"Chromium\";v=\"148\", \"Microsoft Edge\";v=\"148\", \"Not/A)Brand\";v=\"99\"",
        "sec-ch-ua-mobile": "?0",
        "sec-ch-ua-platform": "\"macOS\"",
        "sec-fetch-dest": "empty",
        "sec-fetch-mode": "cors",
        "sec-fetch-site": "same-site",
        "cookie": "粘贴你自己的完整cookie",
        "Referer": "https://www.fenbi.com/"
      },
      "method": "GET"
    });

    const data = await res.json();
    // 👇 这里严格适配你给的接口结构 [[{},{}]]
    const list = data?.[0] || [];
    return list.map(item => item.name).join('、');
  } catch (err) {
    console.warn(`获取考点失败 ID:${questionId}`, err);
    return "无考点";
  }
}
